// in order of precedence from lowest to highest

expr          -> equality
equality      -> comparison (("==" | "!=") comparison)*
comparison    -> term ((">" | ">=" | "<" | "<=") term)*
term          -> factor (("+" | "-") factor)*
factor        -> unary (("*" | "/") unary)*
unary         -> ("-" | "not") unary
              | primary
primary       -> NUMBER | STRING
              | "True" | "False" | "None"
              | "(" expr ")"
